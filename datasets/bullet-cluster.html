WCS & Pixel Scale (quick check)

# (venv) pip install astropy
python - <<'PY'
from astropy.io import fits
from astropy.wcs import WCS
from astropy.wcs.utils import proj_plane_pixel_scales
import glob, os
for f in glob.glob(os.path.expanduser('~/bullet_data/**/*.fits'), recursive=True):
    with fits.open(f) as hdul:
        hdr = hdul[0].header
        try:
            w = WCS(hdr if 'SCI' not in hdul else hdul['SCI'].header)
            pix = proj_plane_pixel_scales(w)[0]*3600.0
            print(os.path.basename(f), 'pixscale=%.5f arcsec/pix' % pix, 'CTYPE1=', w.wcs.ctype[0])
        except Exception as e:
            print(os.path.basename(f), 'WCS: n/a', e)
PY
